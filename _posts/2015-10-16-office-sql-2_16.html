-- Discovers software titles
SELECT DISTINCT sys.Name0, arp.DisplayName00 AS OfficeEdition, arp.Version00 AS OfficeVersion
FROM            System_DISC AS sys LEFT OUTER JOIN
                         Add_Remove_Programs_DATA AS arp ON sys.ItemKey = arp.MachineID
WHERE        (arp.DisplayName00 LIKE 'Microsoft Office %2013%')
