void insertion(int a[], int n)
{
  int i,j,value;
  for(i = 1; i< n; i++) //selecting numbers from second place till end
  {
    value = a[i];
    j = i-1;
    //Insert value=a[i] in the sorted array a[0,1,2....i-1]
    while (j >= 0 and a[j] > value )
    {
      a[j+1]=a[j]; //Shift bigger numbers to right
      j=j-1;
    }
    a[j+1]= value;
  }
}
